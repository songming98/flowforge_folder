<script>
    var cells = document.getElementsByClassName('tg-0lax');

  for (var i = 0; i < cells.length; i++) {
    var cell = cells[i];
    var cellValue = cell.innerText.trim();   // var cellValue = parseFloat(cell.innerText);

    if (cellValue === 'down' || cellValue === 'lost') 
    {
      cell.classList.add('high-value');
    } 
    else if (cellValue === 'run') 
    {
      cell.classList.add('low-value');
    } 
    else 
    {
      cell.classList.add('other-value');
    }
  }
</script>
